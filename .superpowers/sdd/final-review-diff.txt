+interface Props {
+  /** Page title (required) */
+  title: string;
+  /** Page description (required) */
+  description: string;
+  /** Social sharing image (optional, defaults to COMPANY.logo) */
+  image?: string;
+  /** Page canonical URL (optional, defaults to current URL) */
+  url?: string;
+  /** Open Graph type (optional, defaults to 'website') */
+  type?: 'website' | 'product' | 'article';
+  /** Prevent search engines from indexing (optional, defaults to false) */
+  noindex?: boolean;
+}
+
+const {
+  title,
+  description,
+  image = COMPANY.logo,
+  url = Astro.url.href,
+  type = 'website',
+  noindex = false
+} = Astro.props;
+
+/**
+ * Convert relative URLs to absolute URLs
+ * If URL starts with '/', prepend the site origin
+ */
+function toAbsoluteUrl(path: string): string {
+  if (!path) return '';
+  if (path.startsWith('http://') || path.startsWith('https://')) {
+    return path;
+  }
+  const origin = Astro.site?.origin ?? 'https://example.com';
+  return path.startsWith('/') ? `${origin}${path}` : `${origin}/${path}`;
+}
+
+const absoluteImage = toAbsoluteUrl(image);
+const absoluteUrl = toAbsoluteUrl(url);
